Program
: StatementList
;

StatementList
: Statement
| StatementList Statement
;

Statement
: BlockStatement
| VariableDeclaration
| ExpressionStatement -> regular statement, arithmatic operations
;

ExpressionStatement
: Expression
| BinaryExpression -> comparison operator | Literal/Identifier <operator> Literal/Identifier
; 

BlockStatement
: '\s{4}' StatementList
;

Expression
: Operator
| Literal
| Identifier
;

Identifier
: Identifier;

Operator
: Operator
;

Literal
: NumericalLiteral
| StringLiteral
| BooleanLiteral
;

NumericalLiteral
: NUMBERS
;

StringLiteral
: STRING
;
